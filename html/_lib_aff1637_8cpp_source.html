41void LibAff1637::gestion(void) {
42 static bool status = false;
43 static unsigned long nextTimeGestion = millis();
44 unsigned long newTime = millis();
45
46 if( newTime < nextTimeGestion ) return;
47
48 if(m_begin == true ) {
49 if( m_blinking == false ) {
50 writeBrightness((m_brightness & 0x7) | (m_status ? 0x08 : 0x00));
51 } else {
52 status = ( status ? false : true );
53 if( status == false ) {
54 writeBrightness(0x00);
55 } else {
56 writeBrightness((m_brightness & 0x7) | (m_status ? 0x08 : 0x00));
57 }
58 }
59 }
60 nextTimeGestion += PERIOD_CLIGNOTEMENT;
61
62 // On évite l'accumulation de traitements le cas écheant
63 while(newTime > nextTimeGestion) nextTimeGestion += PERIOD_CLIGNOTEMENT;

181bool LibAff1637::writeByte(unsigned char byte) {
182 unsigned char ack;
183
184 // on ecrit les 8 bits
185 for(unsigned char i = 0; i<8; i++) {
186 pinMode(m_pinClk, OUTPUT); // CLK low
187 bitDelay();
188 pinMode(m_pinDIO, (byte & 0x01 ? INPUT : OUTPUT));
189 bitDelay();
190 pinMode(m_pinClk, INPUT); // CLK high
191 bitDelay();
192 byte = byte >> 1;
193 }
194
195 // Wait for acknowledge
196 pinMode(m_pinClk, OUTPUT); // CLK low
197 pinMode(m_pinDIO, INPUT);
198 bitDelay();
199 pinMode(m_pinClk, INPUT); // CLK high
200 bitDelay();
201 ack = digitalRead(m_pinDIO);
202 if (ack == 0)
203 pinMode(m_pinDIO, OUTPUT);
204
205 bitDelay();
206 pinMode(m_pinClk, OUTPUT); // CLK low
207 bitDelay();
208
209 return ack;
210}

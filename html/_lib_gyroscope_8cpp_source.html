52void LibGyroscope::gestion(void) {
53 if(m_begin == false ) return;
54
55 static unsigned long nextTimeGestion = millis(); // Variable persistante
56 static unsigned long nextTimeDisplay = millis(); // Variable persistante
57
58 float ypr[3]; // [yaw, pitch, roll] Yaw/Pitch/Roll container and gravity vector
59
60 unsigned long newTime = millis();
61
62 if( newTime > nextTimeGestion ) {
63 if (mpu.dmpGetCurrentFIFOPacket(m_FIFOBuffer)) { // Get the Latest packet
64 mpu.dmpGetQuaternion(&q, m_FIFOBuffer);
65 mpu.dmpGetGravity(&gravity, &q);
66 mpu.dmpGetYawPitchRoll(ypr, &q, &gravity);
67
68 m_angleX = (ypr[0] * 180/M_PI);
69 m_angleY = (ypr[1] * 180/M_PI);
70 m_angleZ = (ypr[2] * 180/M_PI);
71
72 nextTimeGestion += PERIODE_GESTION;
73 }
74 }
75
76 if( newTime > nextTimeDisplay ) {
77 displayAngle();
78 nextTimeDisplay += PERIODE_UPDATE_DISPLAY;
79 }
80
81 // On évite l'accumulation de traitements le cas écheant
82 while(newTime > nextTimeGestion) nextTimeGestion += PERIODE_GESTION;
83 while(newTime > nextTimeDisplay) nextTimeDisplay += PERIODE_UPDATE_DISPLAY;
84}

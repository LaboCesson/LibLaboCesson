64void LibMcp23008::pinMode( unsigned char pin, unsigned char mode) {
65 if ((pin < 1) || (pin > 8)) return;
66
67 unsigned char mask = 1<<(pin-1);
68 unsigned char dir = readRegister(MCP23008_IODIR);
69 unsigned char olddir = dir;
70 unsigned char pup = readRegister(MCP23008_GPPU);
71 unsigned char oldpup = pup;
72
73 switch(mode) {
74 case INPUT : dir |= mask; pup &= ~mask; break;
75 case INPUT_PULLUP : dir |= mask; pup |= mask; break;
76 case OUTPUT : dir &= ~mask; break;
77 default : return;
78 }

80 if( pup != oldpup ) writeRegister(MCP23008_GPPU, pup);

31void LibMoteur::gestion(void) {
32 if( m_pwmMode == true ) return;
33 static unsigned long timeNext = millis();
34 unsigned long time = millis();
35 static unsigned char nbPas = 0;
36 if( time >= timeNext ) {
37 digitalWrite(m_enableGauche, (m_nbPasEnA>nbPas ? 1 : 0) ) ;
38 digitalWrite(m_enableDroite, (m_nbPasEnB>nbPas ? 1 : 0) ) ;
39 nbPas++;
40 if(nbPas == NB_PAS_GESTION_MOTEUR) nbPas=0;
